// âš™ï¸ Razorpay Webhook Secret (Same for Dev and Prod)
const RAZORPAY_WEBHOOK_SECRET = "CWsecret2025@123";

// âœ³ï¸ Add this just once at the top (to allow raw body for webhook)
app.use("/razorpay-webhook", express.raw({ type: "application/json" }));

// âœ… Webhook endpoint for both dev & prod
app.post("/razorpay-webhook", (req, res) => {
  const signature = req.headers["x-razorpay-signature"];

  try {
    const body = req.body;
    const expectedSignature = crypto
      .createHmac("sha256", RAZORPAY_WEBHOOK_SECRET)
      .update(body)
      .digest("hex");

    if (expectedSignature === signature) {
      const event = JSON.parse(body.toString());
      console.log("âœ… Razorpay Webhook Verified:", event.event);

      if (event.event === "payment.captured") {
        const payment = event.payload.payment.entity;
        console.log("ğŸ’° Payment Captured:", payment);

        // ğŸ”’ Optional: Update Firestore or DB
        /*
        const userEmail = payment.email;
        await db.collection('users').doc(userEmail).update({
          subscriptionStatus: 'active',
          paymentId: payment.id,
          paidAt: new Date()
        });
        */

        res.status(200).send("Webhook received and verified");
      } else {
        console.log("â„¹ï¸ Event not handled:", event.event);
        res.status(200).send("Event not handled");
      }
    } else {
      console.warn("âŒ Invalid webhook signature");
      res.status(400).send("Invalid signature");
    }
  } catch (err) {
    console.error("ğŸš¨ Error processing webhook:", err);
    res.status(500).send("Internal server error");
  }
});
